# a++、++a的用途

```c

    int a = 21;
    int c;
    // a 的值在赋值之前不会自增
    c = (a++)+2;
    // 上面这行代码可以理解为下面这2行代码，括号可以删掉，只是为了方便理解区分
    // c = a+2; >> 23
    // a = a + 1; >> 22
    printf("%d\n",c);
    printf("%d\n",a);

    a = 21;
    // 表达式计算之后，a 的值增加 1
    // a 的值在赋值之前自增
    c = (++a)+2;
    // 上面这行代码，先运行++a，再运行c = a+2
    //a = a + 1;
    //c = a + 2;

    printf("%d\n",c);
    printf("%d\n",a);

    //---------------------------------------------------------------
    int val2 = 5;
    printf("%d,%d,%d",val2,val2+2,val2+3); //从右往左计算，先算val2+3 = 8，再算val2+2 = 7，最后算val2 + 0 = 5
```
