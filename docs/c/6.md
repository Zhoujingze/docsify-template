```c
#include<stdio.h>
int main(){
    int x=10,y=9,a,b,c;
    a = (x--==y++) ? x-- : y++ ;
    //先判断（x--==y++），即x=10与y=9,判断完成后再将y++(y=9+1)后的值代入y++(y=10+1)
    b = x++;
    //上面的已判断完成后，直接将x--(x=10-1)代入即可//
    c = y ;
    printf("%d\n%d\n%d",a,b,c);
}
```
