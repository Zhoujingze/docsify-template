//实验任务1  编程求10-100满足各位上数的乘积大于各位上数的和的所有数，并将结果每行五个的形式输出
#include <stdio.h>
int main()
{
    int a, b, i;
    int count = 0;
    for (i = 10; i <= 100; i++)
    {
        a = i / 10;
        b = i % 10;
        if ((a * b) > (a + b))
        {
            printf("%d ", i);
            count++;
            if (count % 5 == 0)
            {
                printf("\n");
            }
        }
    }
    return 0;
}

//实验任务3  一元钱的兑换方案
#include <stdio.h>
int main()
{
    int i, j, k;
    for (i = 0; i <= 10; i++) // i为一角钱兑换个数，范围为0-10
    {
        for (j = 0; j <= 5; j++) // j为2角钱兑换个数，范围为0-5
        {
            for (k = 0; k <= 2; k++) // k为5角钱兑换个数，范围为0-2
            {
                if (i + 2 * j + 5 * k == 10) //兑换和为10角
                {
                    printf("1元钱可兑换为 ：%d个一角，%d个2角，%d个5角\n", i, j, k); //输出兑换方案
                    printf("\n");
                }
            }
        }
    }
    return 0;
}

{
