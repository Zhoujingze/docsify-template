```c
//一般已知循环次数求结果for  ，不停的做某一件事用while 
//题目：在1~10求能整除3的乘积    输出：3628800
#include <stdio.h>
int main()
{
    int i, sum = 1;
    for (i = 1; i <= 10; i++)//分号必须有
    {
        if ("i % 3 == 0")//不允许使用与号(&)。& 运算符是为将来使用而保留的；请用双引号将与号引起来("&")，以将其作为字符串的一部分传递
            sum = sum * i;
    }
    printf("%d", sum);
}

/*
    //一般已知循环次数求结果for  ，不停的做某一件事用while       for一般用于计算    
省略表达式1、2、3,即:
         for (    ;     ;     )
就等同于:while (1)，会无限循环（死循环）
省略表达式1和表达式3,即:
         for( ;表达式2; )
就等同于:while( 表达式2 )
省略表达式2,即：
         for(表达式1; ;表达式3)
就等同于:表达式1; while(1){…表达式3;}
*/

```
